║                         ЗАПРОС ПОЛЬЗОВАТЕЛЯ                                ║
╚══════════════════════════════════════════════════════════════════════════════╝
                                     │

                          ┌──────────────────┐

                          │   audio/game      │
                          └────────┬─────────┘
                                   │ encode() → число-вектор
                                   ▼
╔══════════════════════════════════════════════════════════════════════════════╗
║  Li CLUSTER  (LiCluster в Li.ts)                                           ║

║   │                          │   │                          │              ║
║   │  Layer 0: [gX₀…gXₙ]     │   │  Layer 0: [gX₀…gXₙ]     │              ║

║   │  Layer N: [gX₀…gXₙ]  ↑  │   │  Layer N: [gX₀…gXₙ]  ↑  │              ║
║   │           grows()     │  │   │           grows()     │  │              ║
║   └────────────┬─────────┘  │   └────────────┬──────────┘  │              ║
║                │             │                │              │              ║
║                └──── Promise.all() ────────────┘              │              ║
║                                   │                           │              ║
║                              buildConsensus()   ←── если resonance < 0.3   ║
║                              mirrorAgreement     →   grow() добавляет слой  ║
╚═══════════════════════════════════╪════════════════════════════════════════╝
                                    │ ClusterResult { consensusOutput, confidence }
                                    ▼
╔══════════════════════════════════════════════════════════════════════════════╗
║  Ω OMEGA  (Omega.ts)                                                        ║
║                                                                              ║
║   let Omega  →  получает ClusterResult от всех Li                           ║
║                                                                              ║
║   Шаг 1:  candidate   = weightedAverage(все Li, по confidence)              ║
║   Шаг 2:  Ω(x)        = candidate                                           ║
║   Шаг 3:  Ω(Ω(x))     = пересчёт кандидата                                 ║
║   Шаг N:  пока |Ω(Ω(x)) − Ω(x)| > threshold → повторить                  ║

                                    │

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 gX НЕЙРОН (внутри каждого слоя Li)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  const gX1 = бит (0 или 1)          — первичное состояние, НИКОГДА не меняется
  const gX2 = gX1 XOR 1              — зеркало, ВСЕГДА противоположность gX1

  input бит → gX1:  primary = true,  score = +1  (резонанс)
  input бит → gX2:  mirror  = true,  score = -1  (анти-резонанс)

  Слой из N нейронов:
    resonance = Σ(score) / N          →  [-1 … +1]
    confidence = f(resonance, activeRate)  →  [0 … 1]


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 ОБУЧЕНИЕ (харвесторы → Li.learn)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Harvester (Python)
       │ REST POST /learn  { key, vector, source }
       ▼
  Li1.learn(key, data)   Li2.learn(key, data)   ← параллельно, Promise.all
       │
       ├─ если key уже есть → EMA обновление (0.9 × old + 0.1 × new)
       └─ если новый        → Map.set()  →  каждые N записей → grow()


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 ТОПОЛОГИЯ (где что живёт)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  [Браузер] ──HTTP/WS──▶ [Vercel / Next.js]  apps/web
                                │ internal HTTP
                                ▼
                         [Render/RunPod]  core/src/server.ts :4000
                          ┌──────────────────────────────┐
                          │  MirrorAI (process RAM)       │
                          │  TextNet · ImageNet · ...     │
                          │  LiCluster (Knowledge Map)    │ ← знания в памяти
                          │  Ω Omega                      │
                          └──────────────┬───────────────┘
                                         │ async, не блокирует
                                         ▼
                                  [Supabase DB]  — логи, история
                                  [Harvesters]   — фоновое обучение


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 СКОРОСТЬ (почему быстро)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  gX activate()     → O(1)  — сравнение одного бита
  Li Knowledge hit  → O(1)  — Map.get() из RAM
  Li process()      → O(layers × neurons), параллельно Li1+Li2
  Ω convergence     → 2-3 итерации, ~5 мс
  Supabase          → НЕ в critical path, пишется после ответа
  Harvesters        → отдельный сервис, НЕ блокируют запрос
